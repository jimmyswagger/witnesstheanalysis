  {
    tt <- seq(start*pi, end*pi, length.out=npoints)
    data.frame(x = center[1] + diameter / 2 * cos(tt), 
               y = center[2] + diameter / 2 * sin(tt))
  }

  }

      geom_segment(aes(x = off13, y = ymin, xend = off13, yend = ymax),colour = line_colour,linetype="solid",size = 0.5) + 
      geom_segment(aes(x = xmin, y = right14d, xend = def13, yend = right14d),colour = line_colour,linetype="solid",size = 0.5) + 
      geom_segment(aes(x = xmin, y = left14d, xend = def13, yend = left14d),colour = line_colour,linetype="solid",size = 0.5) + 
      geom_segment(aes(x = xmax, y = right14d, xend = off13, yend = right14d),colour = line_colour,linetype="solid",size = 0.5) +

      geom_segment(aes(x =glxed, y =  glyst, xend =glxed, yend = glyed),colour = goalb_colour, size = 1) +
      geom_segment(aes(x = xmin, y = rhalfsp, xend = xmax, yend = rhalfsp),colour = "black",linetype="dotted",size = 0.5) +
      geom_segment(aes(x = xmin, y = slwing, xend = xmax, yend = slwing),colour = "black",linetype="dotted",size = 0.5) +
      geom_segment(aes(x = xmin, y = rhalfsp1, xend = xmax, yend = rhalfsp1),colour = "black",linetype="dotted",size = 0.5) +
      geom_segment(aes(x = xmin, y = slwing1, xend = xmax, yend = slwing1),colour = "black",linetype="dotted",size = 0.5) +
      geom_segment(aes(x = 72, y = ymin, xend = 72, yend = ymax),colour = line_colour,linetype="dotted",size = 0.5) +

 geom_segment(aes(x = 11, y = ymin, xend = 11, yend = ymax),colour = line_colour,linetype="dotted",size = 0.5) +
